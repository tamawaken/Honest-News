// Theme toggle (persisted)
(function initTheme() {
  const root = document.documentElement;
  const key = "hn-theme";
  const btn = document.getElementById("themeToggle");
  const saved = localStorage.getItem(key);
  if (saved === "light" || saved === "dark") {
    root.setAttribute("data-theme", saved);
    if (btn) btn.textContent = saved === "dark" ? "🌙" : "☀️";
  }
  if (btn) {
    btn.addEventListener("click", () => {
      const next = root.getAttribute("data-theme") === "dark" ? "light" : "dark";
      root.setAttribute("data-theme", next);
      localStorage.setItem(key, next);
      btn.textContent = next === "dark" ? "🌙" : "☀️";
    });
  }
})();


// Live metrics
async function loadStatus() {
  try {
    const r = await fetch("/status.json", { cache: "no-store" });
    const s = await r.json();
    set("#m-articles", fmt(s.articles_analyzed));
    set("#m-signed", fmt(s.signed_reports));
    set("#m-sources", fmt(s.sources_tracked));
    set("#m-updated", new Date(s.updated_at).toLocaleString());
  } catch {
    // keep placeholders
  }
}
setInterval(loadStatus, 12000);
loadStatus();


// Articles
async function loadArticles() {
  try {
    const res = await fetch("/manifest.json", { cache: "no-store" });
    const manifest = await res.json(); // { items:[{title,date,slug,bias,topic,source,path}] }
    const wrap = qs("#articles");
    wrap.innerHTML = "";
    (manifest.items || []).forEach(item => {
      const card = el("article", "card");
      const h = el("h3"); h.textContent = item.title;
      const meta = el("div", "meta");
      meta.textContent = `${item.source} • ${new Date(item.date).toLocaleDateString()} • ${item.topic}`;
      const tags = el("div", "tags");
      const biasTag = el("span", `tag ${item.bias === "high" ? "bias-high" : ""}`);
      biasTag.textContent = `bias: ${item.bias}`;
      const sigTag = el("span", "tag"); sigTag.textContent = "signed";
      tags.append(biasTag, sigTag);
      const row = el("div");
      const a = el("a", "btn btn-outline");
      a.textContent = "Open analysis";
      a.href = item.path; // direct link to JSON; your /verify.html can load this too
      a.target = "_blank"; a.rel = "noopener";
      row.appendChild(a);
      card.append(h, meta, tags, row);
      wrap.appendChild(card);
    });
  } catch (e) {
    // If manifest missing, show friendly note
    qs("#articles").innerHTML = `<div class="card"><h3>No analyses yet</h3><p class="meta">Add items to <code>/manifest.json</code> and place signed reports under <code>/articles/YYYY-MM-DD/slug.json</code>.</p></div>`;
  }
}
loadArticles();

  );
  window.location.href = `mailto:hello@boonmind.io?subject=${subject}&body=${body}`;
  set("#betaMsg", "Thanks! We opened your email client so you can send the request.");
});


// tiny helpers
function qs(s){return document.querySelector(s)}
function el(tag, cls){const n=document.createElement(tag); if(cls) n.className=cls; return n}
function set(sel, val){const n=qs(sel); if(n) n.textContent = val}
function fmt(n){ if(n==null) return "—"; if(n<1000) return String(n); if(n<1e6) return (n/1e3).toFixed(1)+"k"; return (n/1e6).toFixed(1)+"M"; }
