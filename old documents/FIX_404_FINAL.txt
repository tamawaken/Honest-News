# Commands to verify and fix 404 - Copy-paste one by one:

# 1. Verify files exist on server:
ssh root@45.130.164.130 "ls -la /var/www/html/honest-news/ | head -10"

# 2. Check current config to see location block:
ssh root@45.130.164.130 "cat /etc/nginx/sites-available/default | grep -A 10 'location /honest-news'"

# 3. Fix with proper try_files for alias:
ssh root@45.130.164.130 << 'ENDSSH'
# Remove any existing honest-news location blocks
sed -i '/location \/honest-news/,/^[[:space:]]*}/d' /etc/nginx/sites-available/default

# Add INSIDE server block (replace the closing brace before } with location + brace)
sed -i '/^    }$/{
    i\
\
    # Honest News static site\
    location /honest-news {\
        alias /var/www/html/honest-news;\
        index index.html;\
        try_files $uri $uri/ =404;\
    }
    d
}' /etc/nginx/sites-available/default

# Add closing brace back
echo "}" >> /etc/nginx/sites-available/default

nginx -t && systemctl reload nginx && echo "✅ Fixed!"
ENDSSH

# 4. Test if files are accessible:
ssh root@45.130.164.130 "curl -I http://localhost/honest-news/"
