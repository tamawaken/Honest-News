# Simple commands to fix Nginx - copy-paste one by one:

# 1. Find which config is active:
ssh root@45.130.164.130 "ls -la /etc/nginx/sites-enabled/ 2>/dev/null || echo 'No sites-enabled'; ls -la /etc/nginx/sites-available/"

# 2. View the actual active config to see server block:
ssh root@45.130. shock130 "grep -A 50 'server {' /etc/nginx/nginx.conf | head -60 || grep -A 50 'server {' /etc/nginx/sites-available/* | head -60"

# 3. Fix it properly (add location inside server block):
ssh root@45.130.164.130 << 'ENDSSH'
CONFIG_FILE="/etc/nginx/sites-available/default"
if [ ! -f "$CONFIG_FILE" ]; then
    # Check nginx.conf directly
    CONFIG_FILE="/etc/nginx/nginx.conf"
fi

# Check if we need to add server block or location block
if ! grep -q "server {" "$CONFIG_FILE"; then
    echo "No server block found! Creating one..."
    cat >> "$CONFIG_FILE" << 'EOFCONFIG'
server {
    listen 80 default_server;
    server_name _;
    
    root /var/www/html;
    index index.html;
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    # Honest News static site
    location /honest-news {
        alias /var/www/html/honest-news;
        index index.html;
        try_files $uri $uri/ /honest-news/index.html;
    }
}
EOFCONFIG
else
    # Server block exists, add location inside it
    if ! grep -q "location /honest-news" "$CONFIG_FILE"; then
        # Insert before closing brace of server block
        sed -i '/^\s*server {/,/^\s*}/{
            /^\s*}/{
                i\
    # Honest News static site\
    location /honest-news {\
        alias /var/www/html/honest-news;\
        index index.html;\
        try_files $uri $uri/ /honest-news/index.html;\
    }
            }
        }' "$CONFIG_FILE"
    fi
fi

nginx -t && systemctl reload nginx && echo "✅ Fixed and reloaded!"
ENDSSH
